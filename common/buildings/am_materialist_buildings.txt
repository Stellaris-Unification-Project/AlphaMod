	base_cap_amount = 3

	category = trade

		NOT = { has_modifier = resort_colony }
		exists = owner
		owner = { is_regular_empire = yes }
	}
	
	destroy_trigger = {
		exists = owner
		or = {
			owner = { is_regular_empire = no }

			}
		}
	}
	
	planet_modifier = {
		job_technician_add = 1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {  not = { has_policy_flag = alp_private_businesses_prohibited }}

			trade_value_add = 2
			job_retailer_add = 1
		}
	}

		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_prohibited }
		}
		modifier = {
			job_service_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_freedom }
		}
		modifier = {
			trade_value_add = 2
			job_retailer_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { not = { has_policy_flag = alp_private_businesses_freedom } }
		}
		modifier = {
			job_technician_add = 1
		}

	}

		category = planet_buildings

		}

			energy = 1
		}

			trigger = { owner = { not = { has_policy_flag = alp_private_businesses_prohibited }}}

		}

			trigger = { owner = { not = { has_policy_flag = alp_private_businesses_prohibited }}}

		}
