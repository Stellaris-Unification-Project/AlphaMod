building_automated_alloyforge = {
	base_buildtime = 180
	base_cap_amount = 3
	category = manufacturing
	
	prerequisites = {
		tech_alloys_2
	}
	
	allow = {
		hidden_trigger = {
			exists = owner
			or = {
				owner = { is_ai = no }
				and = {
					owner = { 
						is_ai = yes 
						has_monthly_income = { resource = minerals value > 49 }
						has_monthly_income = { resource = alloys value < 200 }
					}
					free_housing > 1
					free_jobs > 5
				}
			}
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
			minerals = 4
		}
		produces = {
			alloys = 3
		}
	}
	
	ai_resource_production = {
		alloys = 2
		trigger = {
			always = yes
		}
	}
}
