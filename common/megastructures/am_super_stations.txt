research_superstructure = {
	entity = "construction_platform_small_entity"
	construction_entity = "construction_platform_small_entity"
	portrait = "GFX_superstructure_research_background"
	entity_offset = { x = -20 y = -20 }
	rotate_to_center = no
	scale_offset = yes
	place_entity_on_planet_plane = yes
	show_galactic_map_icon = no
	
	build_time = 720
	resources = {
		category = megastructures
		cost = {
			minerals = 1000
			influence = 25
			sr_electronics = 100
		}
		produces = {
			physics_research = 5
			engineering_research = 5
			society_research = 5
		}
		upkeep = {
			energy = 2
			sr_electronics = 1
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					resource_stockpile_compare = {
						resource = sr_electronics
						value <= 0
					}
				}
			}
			energy = 8
		}
	}

	prerequisites = { "tech_space_science_3" }
	show_prereqs = yes
	
	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_an_orbital_research_deposit"
			any_system_planet = {
				has_deposit_for = shipclass_research_station
			}
		}
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}		
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			has_no_non_gate_megastructure = yes
			NOT = {
				any_system_megastructure = {
					OR = {
						is_megastructure_type = research_superstructure
						is_megastructure_type = solarfarm_superstructure
						is_megastructure_type = foundry_superstructure
						is_megastructure_type = hydroponics_superstructure
						is_megastructure_type = shipyard_superstructure
					}
				}
			}
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_an_orbital_research_deposit"
				has_deposit_for = shipclass_research_station
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOR = {
					has_planet_flag = megastructure
					has_planet_flag = has_megastructure
				}
			}
		}
	}

	ai_weight = {
		factor = 10
		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					owner = { is_same_value = from } 
				}
			}
		}
		modifier = {
			factor = 2.0
			from = {
				OR = {
					has_ethic = ethic_materialist
					has_ethic = ethic_fanatic_materialist
				}			
			}
		}
		modifier = {
			factor = 0
			from = {
				OR = {
					resource_stockpile_compare = {
						resource = minerals
						value <= 2500
					}
					resource_stockpile_compare = {
						resource = influence
						value <= 500
					}
					has_monthly_income = { resource = energy value < 25 }
					has_monthly_income = { resource = sr_electronics value < 1 }
				}			
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
	}
}
