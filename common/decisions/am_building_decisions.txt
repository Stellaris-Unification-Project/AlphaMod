			influence = 100
			sr_wealth = 1000
		}
	}

	potential = {
		not = { has_planet_flag = pf_hide_decisions_wealth }
		not = { has_planet_flag = pf_hide_decisions_pop }
		not = { has_planet_flag = pf_filter_vanilla }
		not = { has_modifier = planet_robotbuilding_boost }
		exists = owner
		owner = { is_gestalt = no }
	}

	allow = {
		has_building = building_robot_assembly_plant
	}

	effect = {
		add_modifier = { modifier = "planet_robotbuilding_boost" days = 3600 }
	}

	ai_weight = {
		weight = 5

		# don't encourage growth if there is no free housing
		modifier = {
			factor = 0
			free_housing < 1
		}
	}
}

decision_paradise_citydomes = {
	owned_planets_only = yes
	# enactment_time = 360
	
	prerequisites = {
		"tech_paradise_citydomes"		
	}
	
	resources = {
		category = decisions
		cost = {
			minerals = 250
		}
		cost = {
			trigger = {
				exists = owner
				owner = { 
					is_gestalt = no
					not = { has_valid_civic = civic_elite_easy_living }
				}
			}
			sr_wealth = 200
			rare_crystals = 50
		}
	}
	
	potential = {
		not = { has_planet_flag = pf_hide_decisions_wealth }
		not = { has_planet_flag = pf_hide_decisions_infra }
		not = { has_planet_flag = pf_filter_vanilla }
		not = { has_planet_flag = pf_filter_districts }
		not = { has_building = building_paradise_citydomes }
		exists = owner
		or = {
			num_buildings = { type = building_paradise_dome value > 0 }
			and = {
				owner = { has_valid_civic = civic_elite_easy_living }
				free_building_slots > 0
			}
		}
	}
	
	allow = {
		or = {
			num_buildings = { type = building_paradise_dome value > 1 }
			owner = { has_valid_civic = civic_elite_easy_living }
			not = { has_building = building_paradise_citydomes }
		}
	}
	
	effect = {
		custom_tooltip = decision_paradise_citydomes_custom
		hidden_effect = {
			remove_building = building_paradise_dome
			remove_building = building_paradise_dome
			add_building = building_paradise_citydomes
			validate_planet_buildings_and_districts = yes 
		}
	}
	
	ai_weight = {
		weight = 1
	}
}
