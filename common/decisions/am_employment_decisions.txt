decision_no_police_garrisons = {
	sound = event_administrative_work
	owned_planets_only = yes
	resources = {
		category = decisions
		cost = {
		}
	}
	
	potential = {
		exists = owner
		NOR = {
			has_planet_flag = pf_hide_decisions_pop
			has_planet_flag = pf_hide_decisions_axe
		}
		NOR = {
			has_planet_flag = pf_filter_vanilla
			has_planet_flag = pf_no_police_garrisons
		}
		any_owned_pop = {
			or = {
				has_job = enforcer
				has_job = colonist
				has_job = security_guard
				has_job = ranger
				has_job = imperial_agent
			}
		}
	}
	effect = {
		set_planet_flag = pf_no_police_garrisons
		every_planet_army = {
			limit = { army_type = defense_army }
			remove_army = yes
		}	
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			factor = 1000
			owner = { is_at_war = no }
		}
	}
}

decision_run_jobs_evaluation = {
	sound = event_administrative_work
	owned_planets_only = yes
	resources = {
		category = decisions
		cost = {
		}
	}
	
	potential = {
		NOR = {
			has_planet_flag = pf_filter_vanilla
			has_planet_flag = pf_did_job_eval
		}
		exists = owner
		owner = { is_ai = no }
	}
	effect = {
		every_owned_pop = {
			limit = {
				NOR = {
					has_job = criminal
					has_job = destitute
				}
				NOR = {
					is_unemployed = yes
					is_pop_category = ruler
				}
			}
			# unemploy_pop = yes
			# clear_pop_category = yes
			unemploy_pop = yes
		}
		# check_planet_employment = yes
		check_planet_employment = yes
		
		set_timed_planet_flag = {
			flag = pf_did_job_eval
			days = 5
		}
	}
	
	ai_weight = {
		weight = 0
	}
}

				# has_tradition = tr_prowess_1

			}

	}
	
	allow = {
		NOR = {
			has_modifier = planet_employee_subsidies

		}

		or = {
			free_jobs < 1

		}

		modifier = {
			factor = 0
			owner = {
				resource_stockpile_compare = {
					resource = sr_wealth
					value < 2000
				}
			}
		}
	}
